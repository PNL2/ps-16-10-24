1.To print the sum of uniques numbers in the array 
            num = prompt("").split(" ");
            emp = [], uni = [], dup = [];
            for (i of num) {
                if (!emp.includes(i)) {
                    emp.push(i)
                }
            }
            for (i of emp) {
                c = 0;
                for (j of num) {
                    if (i == j) {
                        c++;
                    }
                }
                if (c == 1)
                    uni.push(i);
            }
            uSum = 0
            for (u of uni) {
                uSum += +u;
            }
            console.log(uSum);


2.To check whether uni sum is more or dup sum is more 
            num = prompt().split(" ");
            emp = [], uni = [], dup = [];
            for (i of num) {
                if (!emp.includes(i)) {
                    emp.push(i)
                }
            }
            for (i of emp) {
                c = 0;
                for (j of num) {
                    if (i == j) {
                        c++;
                    }
                }
                if (c > 1)
                    dup.push(i);
                else if (c == 1)
                    uni.push(i);
            }
            uSum = 0, dSum = 0;
            for (u of uni) {
                uSum += +u;
            }
            for (d of dup) {
                dSum += +d;
            }
            (uSum > dSum) ? console.log("Uniques sum is more") :
            (dSum > uSum) ? console.log("Duplicates sum is more") :
                            console.log("Both sum are same");
